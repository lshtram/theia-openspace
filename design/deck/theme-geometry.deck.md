```typescript
interface GeometryTheme {
  name: 'geometry-warm';
  version: string;
  palette: {
    background: '#fdfaf6';
    text:       '#1a1a2e';
    orange:     '#f97316';
    amber:      '#f59e0b';
    teal:       '#0f766e';
    tealLight:  '#14b8a6';
    rose:       '#e11d48';
  };
  fonts: {
    heading: "'Outfit', 'DM Sans', 'Inter', sans-serif";
    body:    "'Inter', system-ui, sans-serif";
  };
}

function applyGeoTheme(theme: GeometryTheme): void {
  const root = document.documentElement;
  root.style.setProperty('--geo-orange', theme.palette.orange);
  root.style.setProperty('--geo-teal',   theme.palette.teal);
}
```
